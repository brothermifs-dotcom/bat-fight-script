local dragging, dragStart, startPos = false, nil, nil
frame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = frame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then dragging = false end
        end)
    end
end)
frame.InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

    Position = UDim2.new(1, -230, 0, 20)
})

local tweenOut = TweenService:Create(notifFrame, TweenInfo.new(0.4, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {
    Position = UDim2.new(1, 250, 0, 20)
})

local function hideNotification()
    if notifVisible then
        notifClosed = true
        notifVisible = false
        tweenOut:Play()
    end
end

local function showNotification()
    notifClosed = false
    notifVisible = true
    notifSound:Play()
    tweenIn:Play()

    task.delay(4, function()
        if not notifClosed then
            hideNotification()
        end
    end)
end

notifButton.MouseButton1Click:Connect(hideNotification)

notifFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        frame.Visible = true
        guiVisible = true
        closedByX = false
        hideNotification()
    end
end)

closeButton.MouseButton1Click:Connect(function()
    frame.Visible = false
    guiVisible = false
    closedByX = true
    showNotification()
end)

UserInputService.InputBegan:Connect(function(input, processed)
    if processed then return end
    if input.KeyCode == Enum.KeyCode.Q and closedByX then
        guiVisible = not guiVisible
        frame.Visible = guiVisible
        if guiVisible then closedByX = false end
        hideNotification()
    end
end)

    end

local sortedNames = {}
for name in pairs(locations) do table.insert(sortedNames, name) end
table.sort(sortedNames, function(a,b) return a:lower() < b:lower() end)

for index, name in ipairs(sortedNames) do
    local pos = locations[name]

    local button = Instance.new("TextButton", scrollingFrame)
    button.Size = UDim2.new(1, -5, 0, 30)
    button.Text = name
    button.BackgroundColor3 = Color3.fromRGB(40,40,40)
    button.TextColor3 = Color3.new(1,1,1)
    button.LayoutOrder = index + 3

    button.MouseButton1Click:Connect(function()
        previousPosition = humanoidRootPart.CFrame
        humanoidRootPart.CFrame = CFrame.new(pos)
        returnButton.Visible = true
    end)
end
