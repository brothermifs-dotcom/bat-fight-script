--// FINAL Teleport GUI Script for AP: Bat Fight

local ALLOWED_GAME_ID = 12470558349
local GAME_LINK = "https://www.roblox.com/games/12470558349/AP-Bat-Fight"

local player = game.Players.LocalPlayer

-- WARNING GUI if wrong game
if game.PlaceId ~= ALLOWED_GAME_ID then
    local screenGui = Instance.new("ScreenGui")
    screenGui.Parent = player:WaitForChild("PlayerGui")

    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(0, 400, 0, 150)
    frame.Position = UDim2.new(0.5, -200, 0.5, -75)
    frame.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
    frame.BorderSizePixel = 0
    frame.Parent = screenGui

    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(1, -20, 0.7, -10)
    label.Position = UDim2.new(0, 10, 0, 10)
    label.Text = "Wrong supported game!\nGo play "..GAME_LINK
    label.TextColor3 = Color3.new(1,1,1)
    label.BackgroundTransparency = 1
    label.Font = Enum.Font.SourceSansBold
    label.TextScaled = true
    label.TextWrapped = true
    label.Parent = frame

    local button = Instance.new("TextButton")
    button.Size = UDim2.new(0.4, 0, 0.2, 0)
    button.Position = UDim2.new(0.3, 0, 0.75, 0)
    button.Text = "OK"
    button.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    button.TextColor3 = Color3.new(1,1,1)
    button.Font = Enum.Font.SourceSansBold
    button.TextScaled = true
    button.Parent = frame

    -- Dragging functionality
    local UserInputService = game:GetService("UserInputService")
    local dragging = false
    local dragStart = nil
    local startPos = nil

    frame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    frame.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement then
            if dragging then
                local delta = input.Position - dragStart
                frame.Position = UDim2.new(
                    startPos.X.Scale,
                    startPos.X.Offset + delta.X,
                    startPos.Y.Scale,
                    startPos.Y.Offset + delta.Y
                )
            end
        end
    end)

    button.MouseButton1Click:Connect(function()
        pcall(function()
            setclipboard(GAME_LINK)
        end)
        screenGui:Destroy()
    end)

    return
end

--// Supported Game GUI
local character = player.Character or player.CharacterAdded:Wait()
local humanoidRootPart = character:WaitForChild("HumanoidRootPart")

-- Teleport locations
local locations = {
    ["Big Smoke Bat"] = Vector3.new(845, 2062, 695),
    ["Hello Kitty Bat"] = Vector3.new(700, 2091, 870),
    ["Glass Bat"] = Vector3.new(482, 2090, 532),
    ["Void Bat"] = Vector3.new(662, 2103, 144),
    ["Home Run Bat"] = Vector3.new(1102, 2067, 691),
    ["Brazil Bat"] = Vector3.new(1012, 2084, 946),
    ["Corrupted Bat"] = Vector3.new(1467, 2085, 789),
    ["Barbed Bat"] = Vector3.new(1087, 2075, 375),
    ["Magma Bat"] = Vector3.new(1158, 2073, 80),
    ["Radioactive Bat"] = Vector3.new(485, 2070, 522)
}

local screenGui = Instance.new("ScreenGui")
screenGui.Parent = player:WaitForChild("PlayerGui")

local frame = Instance.new("Frame")
frame.Size = UDim2.new(0, 250, 0, 330)
frame.Position = UDim2.new(0, 20, 0.5, -165)
frame.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
frame.Parent = screenGui

local UserInputService = game:GetService("UserInputService")

-- Dragging functionality
local dragging = false
local dragStart = nil
local startPos = nil
frame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = frame.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

frame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        if dragging then
            local delta = input.Position - dragStart
            frame.Position = UDim2.new(
                startPos.X.Scale,
                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
            )
        end
    end
end)

-- Title bar
local titleFrame = Instance.new("Frame")
titleFrame.Size = UDim2.new(1, 0, 0, 30)
titleFrame.Position = UDim2.new(0, 0, 0, 0)
titleFrame.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
titleFrame.Parent = frame

local titleLabel = Instance.new("TextLabel")
titleLabel.Size = UDim2.new(1, -35, 1, 0)
titleLabel.Position = UDim2.new(0, 5, 0, 0)
titleLabel.Text = "[UPD🎮] AP: Bat Fight"
titleLabel.TextColor3 = Color3.new(1,1,1)
titleLabel.BackgroundTransparency = 1
titleLabel.Font = Enum.Font.SourceSansBold
titleLabel.TextScaled = true
titleLabel.TextXAlignment = Enum.TextXAlignment.Left
titleLabel.Parent = titleFrame

-- Close button X
local closeButton = Instance.new("TextButton")
closeButton.Size = UDim2.new(0, 25, 0, 25)
closeButton.Position = UDim2.new(1, -30, 0, 2.5)
closeButton.Text = "X"
closeButton.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
closeButton.TextColor3 = Color3.new(1,1,1)
closeButton.Font = Enum.Font.SourceSansBold
closeButton.TextScaled = true
closeButton.Parent = titleFrame

local guiVisible = true
local closedByX = false

closeButton.MouseButton1Click:Connect(function()
    frame.Visible = false
    guiVisible = false
    closedByX = true
end)

-- Toggle GUI with Q (only after X closed)
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    if input.KeyCode == Enum.KeyCode.Q and closedByX then
        guiVisible = not guiVisible
        frame.Visible = guiVisible
        if guiVisible then
            closedByX = false
        end
    end
end)

-- Scrolling frame
local scrollingFrame = Instance.new("ScrollingFrame")
scrollingFrame.Size = UDim2.new(1, -10, 1, -35)
scrollingFrame.Position = UDim2.new(0, 5, 0, 35)
scrollingFrame.ScrollBarThickness = 6
scrollingFrame.BackgroundTransparency = 1
scrollingFrame.Parent = frame

local layout = Instance.new("UIListLayout")
layout.Padding = UDim.new(0, 5)
layout.SortOrder = Enum.SortOrder.LayoutOrder
layout.Parent = scrollingFrame

-- Alphabetical buttons
local sortedNames = {}
for name in pairs(locations) do
    table.insert(sortedNames, name)
end
table.sort(sortedNames, function(a,b) return a:lower() < b:lower() end)

for index, name in ipairs(sortedNames) do
    local position = locations[name]
    local button = Instance.new("TextButton")
    button.Size = UDim2.new(1, -5, 0, 30)
    button.Text = name
    button.BackgroundColor3 = Color3.fromRGB(40,40,40)
    button.TextColor3 = Color3.new(1,1,1)
    button.LayoutOrder = index
    button.Parent = scrollingFrame

    button.MouseButton1Click:Connect(function()
        humanoidRootPart.CFrame = CFrame.new(position)
    end)
end

-- Dynamic scroll
scrollingFrame.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y)
layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
    scrollingFrame.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y)
end)

-- Keep GUI on respawn
local function keepGUI(gui)
    local player = game.Players.LocalPlayer
    player.CharacterAdded:Connect(function()
        gui.Parent = player:WaitForChild("PlayerGui")
    end)
end
keepGUI(screenGui)
